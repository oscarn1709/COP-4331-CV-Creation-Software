# SQL commands to create and populate the MySQL database for Group Project
#
# delete the database if it already exists
drop database if exists CV_DB;

# create a new database named eventdb
create database CV_DB;

# switch to the new database
use CV_DB;


CREATE TABLE user (
    uid INTEGER NOT NULL AUTO_INCREMENT,
    username VARCHAR(255) NOT NULL,
    password VARCHAR(50) NOT NULL,
    name VARCHAR(50),
    address VARCHAR(70),
    email VARCHAR(60),
    phone VARCHAR(12),
    skills VARCHAR (1000),
    optTitle VARCHAR (25),
    optional VARCHAR (1000),
    university1 VARCHAR(70),
    degree1 VARCHAR(50),
    major1 VARCHAR(50),
    gpa1 VARCHAR(5),
    grad1 VARCHAR (20),
    university2 VARCHAR(70),
    degree2 VARCHAR(50),
    major2 VARCHAR(50),
    gpa2 VARCHAR(5),
    grad2 VARCHAR(20),
    company1 VARCHAR(70),
    title1 VARCHAR(50),
    begin1 VARCHAR(20),
    end1 VARCHAR(20),
    resp1 VARCHAR(1000),
    company2 VARCHAR(70),
    title2 VARCHAR(50),
    begin2 VARCHAR(20),
    end2 VARCHAR(20),
    resp2 VARCHAR(1000),
    company3 VARCHAR(70),
    title3 VARCHAR(50),
    begin3 VARCHAR(20),
    end3 VARCHAR(20),
    resp3 VARCHAR(1000),
    image LONGBLOB,
    PRIMARY KEY (uid)
);
